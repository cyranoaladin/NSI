\section*{Statut}
TP papier : aucun code Python attendu ; le livrable est une trace écrite vérifiable.

\section*{Donnée fournie}
\texttt{tableau=[4,9,18,23,37,41], cible=37} ; \texttt{pièces=[10,5,2,1], montant=28} ; \texttt{voisins=[rouge:1.2, bleu:2.0, rouge:2.4]}.

\section*{Travail demandé}
\begin{enumerate}[nosep]
  \item Préparer la donnée et nommer les champs utiles.
  \item Calculer le milieu puis réduire l'intervalle.
  \item Montrer que le variant droite-gauche diminue.
  \item Tester le cas limite \texttt{cible absente}.
  \item Produire le livrable : milieux $18$ puis $37$ $\rightarrow$ indice $4$.
\end{enumerate}

\section*{Barème}
\begin{bareme}
  {Donnée préparée} & 2 \\
  {Méthode principale} & 3 \\
  {Résultat (indice 4)} & 3 \\
  {Cas limite \texttt{cible absente}} & 2 \\
\end{bareme}

\section*{Corrigé}
\begin{enumerate}[nosep,label=Q\arabic*.]
  \item donnée préparée, champs nommés.
  \item milieux $18$ puis $37$ $\rightarrow$ indice $4$.
  \item $V = \text{droite} - \text{gauche} + 1$ décroît de $6$ à $3$ $\rightarrow$ terminaison.
  \item \texttt{cible absente} traité sans ambiguïté.
\end{enumerate}
